bool hormodular::ModularRobot::reset ( )

Resets the initial state of the robot.

It should be used after run() if the robot is to be run again.

Returns:
True if completed successfully, false otherwise 
{
    //-- Reset robot
    if (!robotInterface->reset())
        return false;

    //-- Initialize joint vector to 0
    joint_values.clear();
    for (int i = 0; i < (int) configParser.getNumModules(); i++)
        joint_values.push_back(0);

    //-- Attach modules
    if (!attachModules())
        return false;

    elapsed_time = 0;

    return true;
}
